package edu.brown.cs.student.main.UtilityClasses;

import edu.brown.cs.student.main.DeveloperObjectClasses.ArrayListCreator;
import edu.brown.cs.student.main.FactoryFailureException;
import java.io.File;
import java.io.FileReader;
import java.io.IOException;
import java.util.ArrayList;

/**
 * A class that is user-facing and creates a level of abstraction by creating Searcher and Parser in
 * here
 */
public class Utility {
  private FileReader reader;
  private ArrayListCreator creator;
  private Parser<ArrayList<String>> parser;
  private Searcher searcher;
  private String header;
  // ed post #99

  /**
   * This Utility constructor is used when the user wants to search through a specific header
   *
   * @param CSVFile the CSV file
   * @param header the header to search through
   * @param value the value to search for
   * @throws IOException
   * @throws FactoryFailureException
   */
  public Utility(File CSVFile, String header, String value)
      throws IOException, FactoryFailureException {
    if (header == null) {
      System.out.println("If you wish to search with a header, please provide one");
    }
    this.header = header;
    this.utility(CSVFile, value);
  }

  /**
   * This constructor is used when the user does not specify a header to search through
   *
   * @param CSVFile the CSV file
   * @param value the value to search for
   * @throws IOException
   * @throws FactoryFailureException
   */
  public Utility(File CSVFile, String value) throws IOException, FactoryFailureException {
    this.header = null;
    this.utility(CSVFile, value);
  }

  /**
   * A method that takes care of most of the functionality by creating instances of Searcher and
   * Parser her rather than in Main
   *
   * @param CSVFile the CSV file
   * @param value the value to search for
   * @throws IOException
   * @throws FactoryFailureException
   */
  private void utility(File CSVFile, String value) throws IOException, FactoryFailureException {
    if (this.fileChecker(CSVFile)) {
      this.reader = new FileReader(CSVFile);
      this.creator = new ArrayListCreator();
      this.parser = new Parser<>(this.reader, this.creator);
      this.searcher = new Searcher(this.parser, this.isSearchingWithHeader());
      System.out.println(this.searcher.search(value, this.header));
    }
  }

  /**
   * Returns whether the user is searching with a header
   *
   * @return
   */
  private boolean isSearchingWithHeader() {
    return this.header != null;
  }

  /**
   * Checks that the provided CSV file is within the data directory. If not it prints out an error
   * message and returns false.
   *
   * @param CSVFile the CSV file
   * @return
   */
  private boolean fileChecker(File CSVFile) {
    String rootPath = "data/";
    String fileName = CSVFile.getName();
    String rootDirectory = fileName.substring(0, 5);
    if (!CSVFile.exists() && !rootPath.equals(rootDirectory)) {
      System.err.println(
          "Unable to use file: file either does not exist or is not in the data directory");
      // System.exit(1);
      return false;
    } else {
      return true;
    }
  }
}
